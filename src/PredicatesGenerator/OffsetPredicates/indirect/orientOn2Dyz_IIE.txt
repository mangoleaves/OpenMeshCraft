// p1 is query point, p2-p3 is the line
// orient2d: (p1 - p3) cross (p2 - p3)

// we calculate determinant
//    | p1 - p3 |
//    | p2 - p3 |
// expand to
//    | b1 + l1/d1 - p3 |
//    | b2 + l2/d2 - p3 |
// transform to
//    | l1 + d1 (b1 - p3) |
//    | l2 + d2 (b2 - p3) |

// -- b1 - p3
b1p3y = b1y - p3y
b1p3z = b1z - p3z

// -- b2 - p3
b2p3y = b2y - p3y
b2p3z = b2z - p3z

// -- d1 * (b1 - p3)
d1_b1p3y = d1 * b1p3y
d1_b1p3z = d1 * b1p3z

// -- d1 * (b1 - p3) + l1
i1y = d1_b1p3y + l1y
i1z = d1_b1p3z + l1z

// -- d2 * (b2 - p3)
d2_b2p3y = d2 * b2p3y
d2_b2p3z = d2 * b2p3z

// -- d2 * (b2 - p3) + l2
i2y = d2_b2p3y + l2y
i2z = d2_b2p3z + l2z

t0 = i1y * i2z
t1 = i1z * i2y
det = t0 - t1
